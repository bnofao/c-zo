## Naming convention (verified)

A `relayMutationField('foo', { inputFields }, { errors, resolve }, { outputFields })` produces:
- input type **`FooInput`**
- result union **`FooResult`**, success member **`FooSuccess`**, payload nested under **`data { ... }`**
- error members named `<TaggedError>Error` (e.g. `OrganizationSlugTaken` → `OrganizationSlugTakenError`)

So a mutation is queried as:
```graphql
mutation ($input: FooInput!) {
  foo(input: $input) {
    ... on FooSuccess { data { <payload fields> } }
    ... on SomeErrorError { message }   # to assert a typed error member
  }
}
```
A denial by `authScopes` surfaces as a top-level `errors[]` entry (not a union member); assert `result.errors` is present. A `node(id:)` guard denial surfaces as `data.node === null`.

## Authz model (verified — for the helpers)

- **Global tier:** `users.role` (string). A role containing `admin` (`'admin'`, `'admin:manager'`, …) is the admin tier. Granted in tests via `UserService.setRole(userId, role)`.
- **Org tier:** membership `role` is a comma-separated string mixing org roles + module roles, e.g. `'org:owner,stock-loc:viewer,stock-loc:manager,stock-loc:admin'`. Org roles: `org:viewer` (member/invitation read), `org:admin` (org update, member CRUD, invitation create/cancel), `org:owner` (+ org delete). Granted via `OrganizationService.updateMemberRole({ id, organizationId, role })` after finding the member with `findFirstMember(orgId, { where: { userId } })`.
- **Stock-location org perms:** roles `stock-loc:viewer` (read), `stock-loc:manager` (create/update), `stock-loc:admin` (delete) — grant all three for full CRUD.

A condensed field reference is in the **Appendix** at the end — copy exact names from there.

```ts
/**
 * Shared E2E harness for the auth module's GraphQL + REST surface.
 * Boots [auth] on a Testcontainers Postgres via bootTestApp and drives the real
 * h3/Yoga fetch handler. Plain vitest describe/it use it via beforeAll/afterAll.
 */
import { dirname, resolve } from 'node:path'
import process from 'node:process'
import { fileURLToPath } from 'node:url'
import { decodeGlobalID } from '@czo/kit/graphql'
import { bootTestApp } from '@czo/kit/testing'
import { Effect, Exit, Scope } from 'effect'
import authModule from '../index'
import { Organization, User } from '../services'

const here = dirname(fileURLToPath(import.meta.url))
const AUTH_MIGRATIONS = resolve(here, '../../migrations')

const GRAPHQL_URL = 'http://localhost/graphql'
const AUTH_URL = 'http://localhost/api/auth'

export interface BootedApp {
  fetch: (req: Request) => Promise<Response>
  runEffect: <A, E>(e: Effect.Effect<A, E, any>) => Promise<A>
  close: () => Promise<void>
}

export interface GqlResult {
  data?: any
  errors?: { message: string, path?: string[], extensions?: Record<string, unknown> }[]
}

export interface SignedUpUser {
  readonly token: string
  readonly userId: number
  /** Synthetic per-actor client IP (sent as x-forwarded-for) so suites never share a rate-limit bucket. */
  readonly ip: string
}

export interface AuthHarness {
  readonly app: BootedApp
  readonly close: () => Promise<void>
  /** POST a GraphQL op; optional bearer token + client ip. */
  readonly gql: (query: string, variables?: Record<string, unknown>, token?: string, ip?: string) => Promise<GqlResult>
  /** REST sign-up; returns bearer token + numeric id + a unique synthetic IP. */
  readonly signUp: (email: string, name: string, password: string) => Promise<SignedUpUser>
  /** REST sign-in; returns the raw Response so callers can read status + Set-Cookie + body. */
  readonly signIn: (email: string, password: string, ip?: string) => Promise<Response>
  /** REST sign-out for a bearer token; returns the raw Response. */
  readonly signOut: (token: string, ip?: string) => Promise<Response>
  /** Grant a GLOBAL role (users.role) — platform/admin tier. */
  readonly grantGlobalRole: (userId: number, role: string) => Promise<void>
  /** Create an org as token's user; returns relay + numeric id. */
  readonly createOrganization: (token: string, name: string, slug: string, ip?: string) => Promise<{ orgGlobalId: string, orgNumericId: number }>
  /** Set a member's full role string in an org (org + module perms), found by userId. */
  readonly setMemberRole: (orgNumericId: number, userId: number, role: string) => Promise<void>
}

export async function bootAuthApp(): Promise<AuthHarness> {
  // eslint-disable-next-line turbo/no-undeclared-env-vars -- test-only secret; auth reads it via Effect Config at boot
  process.env.AUTH_SECRET = 'x'.repeat(40)
  // eslint-disable-next-line turbo/no-undeclared-env-vars -- test-only app id; auth reads it via Effect Config at boot
  process.env.AUTH_APP = 'test'

  const scope = await Effect.runPromise(Scope.make())
  const app = (await Effect.runPromise(
    bootTestApp({ modules: [authModule], migrations: [AUTH_MIGRATIONS] })
      .pipe(Effect.provideService(Scope.Scope, scope)),
  )) as BootedApp

  let signUpCount = 0

  const gql: AuthHarness['gql'] = async (query, variables = {}, token, ip) => {
    const res = await app.fetch(new Request(GRAPHQL_URL, {
      method: 'POST',
      headers: {
        'content-type': 'application/json',
        ...(token ? { authorization: `Bearer ${token}` } : {}),
        ...(ip ? { 'x-forwarded-for': ip } : {}),
      },
      body: JSON.stringify({ query, variables }),
    }))
    return res.json() as Promise<GqlResult>
  }

  const signUp: AuthHarness['signUp'] = async (email, name, password) => {
    const res = await app.fetch(new Request(`${AUTH_URL}/sign-up`, {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ email, name, password }),
    }))
    const body = (await res.json()) as { token?: string }
    if (!res.ok || !body.token)
      throw new Error(`sign-up failed (${res.status}): ${JSON.stringify(body)}`)
    signUpCount += 1
    return { token: body.token, userId: signUpCount, ip: `10.0.0.${signUpCount}` }
  }

  const signIn: AuthHarness['signIn'] = (email, password, ip) =>
    app.fetch(new Request(`${AUTH_URL}/sign-in`, {
      method: 'POST',
      headers: { 'content-type': 'application/json', ...(ip ? { 'x-forwarded-for': ip } : {}) },
      body: JSON.stringify({ email, password }),
    }))

  const signOut: AuthHarness['signOut'] = (token, ip) =>
    app.fetch(new Request(`${AUTH_URL}/sign-out`, {
      method: 'POST',
      headers: { authorization: `Bearer ${token}`, ...(ip ? { 'x-forwarded-for': ip } : {}) },
    }))

  const grantGlobalRole: AuthHarness['grantGlobalRole'] = (userId, role) =>
    app.runEffect(Effect.gen(function* () {
      const users = yield* User.UserService
      yield* users.setRole(userId, role)
    })).then(() => undefined)

  const createOrganization: AuthHarness['createOrganization'] = async (token, name, slug, ip) => {
    const created = await gql(
      `mutation ($input: CreateOrganizationInput!) {
        createOrganization(input: $input) {
          ... on CreateOrganizationSuccess { data { organization { id } } }
        }
      }`,
      { input: { name, slug } },
      token,
      ip,
    )
    if (created.errors)
      throw new Error(`createOrganization failed: ${JSON.stringify(created.errors)}`)
    const orgGlobalId: string | undefined = created.data?.createOrganization?.data?.organization?.id
    if (!orgGlobalId)
      throw new Error(`createOrganization returned no org id: ${JSON.stringify(created.data)}`)
    return { orgGlobalId, orgNumericId: Number(decodeGlobalID(orgGlobalId).id) }
  }

  const setMemberRole: AuthHarness['setMemberRole'] = (orgNumericId, userId, role) =>
    app.runEffect(Effect.gen(function* () {
      const org = yield* Organization.OrganizationService
      const member = yield* org.findFirstMember(orgNumericId, { where: { userId } })
      yield* org.updateMemberRole({ id: member.id, organizationId: orgNumericId, role })
    })).then(() => undefined)

  const close = async () => {
    await app.close()
    await Effect.runPromise(Scope.close(scope, Exit.void))
  }

  return { app, close, gql, signUp, signIn, signOut, grantGlobalRole, createOrganization, setMemberRole }
}
```

import { bootAuthApp, type AuthHarness } from './harness'

describe('REST auth (E2E)', () => {

  beforeAll(async () => { h = await bootAuthApp() }, 120_000)

      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ email: 'dupe@ex.com', name: 'Dupe2', password: 'password123!' }),

  it('signs in with correct credentials (sets cookie + returns token)', async () => {

    await h.signUp('wrongpw@ex.com', 'Wrong', 'password123!')
    const res = await h.signIn('wrongpw@ex.com', 'nope-nope-nope', '10.9.9.2')
    expect(res.status).toBe(401)
  })

  it('signs out (200)', async () => {
    const u = await h.signUp('out@ex.com', 'Out', 'password123!')
    const res = await h.signOut(u.token, u.ip)
    expect(res.status).toBe(200)
  })
})

**Files:**

- [ ] **Step 1: Write the suite**

Covers: create (happy), org-permission tier (member with `org:admin` can update; plain member cannot → denial), invitations (invite → accept), member role update, queries, and **cross-org denial** (a member of org A cannot read org B's members). Uses the Appendix field names.

```ts
import { afterAll, beforeAll, describe, expect, it } from 'vitest'
import { bootAuthApp, type AuthHarness } from './harness'

describe('Organization (E2E)', () => {
  let h: AuthHarness
  beforeAll(async () => { h = await bootAuthApp() }, 120_000)
  afterAll(() => h.close())

  it('creates an organization (creator becomes owner)', async () => {
    const owner = await h.signUp('org-owner@ex.com', 'Owner', 'password123!')
    const { orgGlobalId } = await h.createOrganization(owner.token, 'Acme', 'acme', owner.ip)
    expect(orgGlobalId).toBeTruthy()
  })

  it('enforces org:update permission on updateOrganization', async () => {
    const owner = await h.signUp('org2-owner@ex.com', 'Owner2', 'password123!')
    const { orgGlobalId, orgNumericId } = await h.createOrganization(owner.token, 'Beta', 'beta', owner.ip)

    // A second user joins as a plain viewer (member:read only, NOT organization:update).
    const viewer = await h.signUp('org2-viewer@ex.com', 'Viewer', 'password123!')
    // owner invites viewer, viewer accepts
    const inv = await h.gql(
      `mutation ($i: InviteMemberInput!) { inviteMember(input: $i) {
        ... on InviteMemberSuccess { data { invitation { id } } } } }`,
      { i: { organizationId: orgGlobalId, email: 'org2-viewer@ex.com', role: 'org:viewer' } },
      owner.token, owner.ip,
    )
    const invId: string = inv.data.inviteMember.data.invitation.id
    const accept = await h.gql(
      `mutation ($i: AcceptInvitationInput!) { acceptInvitation(input: $i) {
        ... on AcceptInvitationSuccess { data { member { id } } } } }`,
      { i: { invitationId: invId } }, viewer.token, viewer.ip,
    )
    expect(accept.errors).toBeUndefined()

    // viewer tries to update the org -> denied (no organization:update perm)
    const denied = await h.gql(
      `mutation ($i: UpdateOrganizationInput!) { updateOrganization(input: $i) {
        ... on UpdateOrganizationSuccess { data { organization { id } } } } }`,
      { i: { id: orgGlobalId, name: 'Beta Renamed' } }, viewer.token, viewer.ip,
    )
    expect(denied.errors).toBeTruthy()
    expect(denied.data?.updateOrganization ?? null).toBeNull()

    // owner CAN update
    const ok = await h.gql(
      `mutation ($i: UpdateOrganizationInput!) { updateOrganization(input: $i) {
        ... on UpdateOrganizationSuccess { data { organization { id name } } } } }`,
      { i: { id: orgGlobalId, name: 'Beta Renamed' } }, owner.token, owner.ip,
    )
    expect(ok.errors).toBeUndefined()
    expect(ok.data.updateOrganization.data.organization.name).toBe('Beta Renamed')
    void orgNumericId
  })

  it('denies cross-org member reads', async () => {
    const a = await h.signUp('cross-a@ex.com', 'A', 'password123!')
    const b = await h.signUp('cross-b@ex.com', 'B', 'password123!')
    const orgA = await h.createOrganization(a.token, 'OrgA', 'org-a', a.ip)
    await h.createOrganization(b.token, 'OrgB', 'org-b', b.ip)
    // b (not a member of orgA) reads orgA members -> denied (member:read perm missing)
    const res = await h.gql(
      `query ($id: ID!) { members(organizationId: $id) { edges { node { id } } } }`,
      { id: orgA.orgGlobalId }, b.token, b.ip,
    )
    expect(res.errors).toBeTruthy()
  })
})
```

import { afterAll, beforeAll, describe, expect, it } from 'vitest'
import { bootAuthApp, type AuthHarness } from './harness'

const CREATE = `mutation ($i: CreateApiKeyInput!) {

    ... on CreateApiKeySuccess { data { apiKey { id name } plain } }
    ... on UnauthenticatedError { message }
  }

describe('API keys (E2E)', () => {
  let h: AuthHarness
  beforeAll(async () => { h = await bootAuthApp() }, 120_000)
  afterAll(() => h.close())

  it('creates a USER-owned api key and returns the plaintext once', async () => {
    const u = await h.signUp('ak-user@ex.com', 'AK', 'password123!')
    const res = await h.gql(CREATE, {

## Task 5: Account-flows suite

**Files:**
- Create: `packages/modules/auth/src/e2e/account.e2e.test.ts`

- [ ] **Step 1: Write the suite**

Covers the authenticated account mutations end-to-end. Token-flows that send an email + require a token (password reset, email verification, email change) can't easily read the token in a black-box E2E, so assert the **request** mutation succeeds (`success: true`) and that `{auth:true}`-gated ones are **denied when unauthenticated**. `changePassword` is fully testable (then sign in with the new password).

```ts
import { afterAll, beforeAll, describe, expect, it } from 'vitest'
import { bootAuthApp, type AuthHarness } from './harness'

describe('Account flows (E2E)', () => {
  let h: AuthHarness
  beforeAll(async () => { h = await bootAuthApp() }, 120_000)
  afterAll(() => h.close())

  it('changePassword then sign-in with the new password', async () => {
    const u = await h.signUp('acct-pw@ex.com', 'PW', 'password123!')
    const res = await h.gql(
      `mutation ($i: ChangePasswordInput!) { changePassword(input: $i) {
        ... on ChangePasswordSuccess { data { success } }
        ... on IncorrectCurrentPasswordError { message } } }`,
      { i: { currentPassword: 'password123!', newPassword: 'newpassword456!' } }, u.token, u.ip,

/** Shared E2E harness for stock-location: boots [auth, stock-location]. */
import { dirname, resolve } from 'node:path'
import process from 'node:process'
import { fileURLToPath } from 'node:url'
import { decodeGlobalID } from '@czo/kit/graphql'
import { bootTestApp } from '@czo/kit/testing'
import { Effect, Exit, Scope } from 'effect'
import authModule from '../../../auth/src/index'
import { Organization, User } from '../../../auth/src/services'
import stockLocationModule from '../index'

const here = dirname(fileURLToPath(import.meta.url))
const AUTH_MIGRATIONS = resolve(here, '../../../auth/migrations')
const SL_MIGRATIONS = resolve(here, '../../migrations')

const GRAPHQL_URL = 'http://localhost/graphql'
const AUTH_URL = 'http://localhost/api/auth'

export interface GqlResult { data?: any, errors?: { message: string }[] }
export interface SignedUpUser { token: string, userId: number, ip: string }

export interface SlHarness {
  app: { fetch: (r: Request) => Promise<Response>, runEffect: <A, E>(e: Effect.Effect<A, E, any>) => Promise<A>, close: () => Promise<void> }
  close: () => Promise<void>
  gql: (q: string, v?: Record<string, unknown>, token?: string, ip?: string) => Promise<GqlResult>
  signUp: (email: string, name: string, password: string) => Promise<SignedUpUser>
  createOrganization: (token: string, name: string, slug: string, ip?: string) => Promise<{ orgGlobalId: string, orgNumericId: number }>
  setMemberRole: (orgNumericId: number, userId: number, role: string) => Promise<void>
}

export async function bootStockLocationApp(): Promise<SlHarness> {
  // eslint-disable-next-line turbo/no-undeclared-env-vars -- test-only secret
  process.env.AUTH_SECRET = 'x'.repeat(40)
  // eslint-disable-next-line turbo/no-undeclared-env-vars -- test-only app id
  process.env.AUTH_APP = 'test'

  const scope = await Effect.runPromise(Scope.make())
  const app = (await Effect.runPromise(
    bootTestApp({ modules: [authModule, stockLocationModule], migrations: [AUTH_MIGRATIONS, SL_MIGRATIONS] })
      .pipe(Effect.provideService(Scope.Scope, scope)),
  )) as SlHarness['app']

  let n = 0
  const gql: SlHarness['gql'] = async (query, variables = {}, token, ip) => {
    const res = await app.fetch(new Request(GRAPHQL_URL, {
      method: 'POST',
      headers: { 'content-type': 'application/json', ...(token ? { authorization: `Bearer ${token}` } : {}), ...(ip ? { 'x-forwarded-for': ip } : {}) },
      body: JSON.stringify({ query, variables }),
    }))
    return res.json() as Promise<GqlResult>
  }
  const signUp: SlHarness['signUp'] = async (email, name, password) => {
    const res = await app.fetch(new Request(`${AUTH_URL}/sign-up`, {
      method: 'POST', headers: { 'content-type': 'application/json' }, body: JSON.stringify({ email, name, password }),
    }))
    const body = (await res.json()) as { token?: string }
    if (!res.ok || !body.token) throw new Error(`sign-up failed (${res.status})`)
    n += 1
    return { token: body.token, userId: n, ip: `10.0.0.${n}` }
  }
  const createOrganization: SlHarness['createOrganization'] = async (token, name, slug, ip) => {
    const created = await gql(
      `mutation ($input: CreateOrganizationInput!) { createOrganization(input: $input) {
        ... on CreateOrganizationSuccess { data { organization { id } } } } }`,
      { input: { name, slug } }, token, ip,
    )
    const id: string | undefined = created.data?.createOrganization?.data?.organization?.id
    if (!id) throw new Error(`createOrganization failed: ${JSON.stringify(created.errors ?? created.data)}`)
    return { orgGlobalId: id, orgNumericId: Number(decodeGlobalID(id).id) }
  }
  const setMemberRole: SlHarness['setMemberRole'] = (orgNumericId, userId, role) =>
    app.runEffect(Effect.gen(function* () {
      const org = yield* Organization.OrganizationService
      const member = yield* org.findFirstMember(orgNumericId, { where: { userId } })
      yield* org.updateMemberRole({ id: member.id, organizationId: orgNumericId, role })
    })).then(() => undefined)
  const close = async () => { await app.close(); await Effect.runPromise(Scope.close(scope, Exit.void)) }
  void User // imported for parity / potential global-role grants
  return { app, close, gql, signUp, createOrganization, setMemberRole }
}
```
