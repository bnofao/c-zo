```ts
/**
 * Crypto primitives for the auth module.
 *
 * RUNTIME ASSUMPTION: Node ≥18 (`node:crypto`). All Node-specific calls live
 * in this file alone — a runtime change (Bun, Deno, Workers, edge) only
 * requires swapping the bodies here for Web Crypto / @oslojs / etc.
 */
import { createHash, randomBytes } from 'node:crypto'

export function sha256Hex(plain: string): string {
  return createHash('sha256').update(plain).digest('hex')
}

export function randomString(length: number, alphabet?: string): string {
  const chars = alphabet ?? 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789'
  const bytes = randomBytes(length)
  let out = ''
  for (let i = 0; i < length; i++) out += chars[bytes[i]! % chars.length]
  return out
}
```

```ts
      authorize: (granted, required, connector = 'AND') =>
        Effect.sync(() => {
          if (!granted) return false
          for (const [resource, actions] of Object.entries(required) as [string, string[]][]) {
            const grantedActions = (granted as Record<string, string[]>)[resource]
            if (!grantedActions) return false
            const hasAll = actions.every(a => grantedActions.includes(a))
            const hasAny = actions.some(a => grantedActions.includes(a))
            if (connector === 'AND' && !hasAll) return false
            if (connector === 'OR' && !hasAny) return false
          }
          return true
        }),
```

```ts
import type { AuthGraphQLSchemaBuilder } from '../../index'
import type { GraphQLContextMap } from '@czo/kit/graphql'
import { DrizzleDb } from '@czo/kit/db/effect'
import { UnauthenticatedError, decodeGlobalID } from '@czo/kit/graphql'
import { Effect } from 'effect'
import { OrganizationService } from '@czo/auth/services/organization'

// ─── API Key Queries ──────────────────────────────────────────────────────────

export function registerApiKeyQueries(builder: AuthGraphQLSchemaBuilder): void {
  // ── apiKey(id) — single API key, ownership-OR-membership guarded ─────────
  builder.queryField('apiKey', t =>
    t.drizzleField({
      type: 'apikeys',
      nullable: true,
      args: {
        id: t.arg.id({ required: true }),
      },
      authScopes: { auth: true },
      resolve: async (query, _root: unknown, args: Record<string, unknown>, ctx: GraphQLContextMap) => {
        const user = ctx.auth?.user
        if (!user)
          throw new UnauthenticatedError()

        const db = await ctx.runEffect(DrizzleDb) as any
        const keyId = Number(decodeGlobalID(String(args.id)).id)
        const key = await db.query.apikeys.findFirst(
          query({ where: (k: any, { eq }: any) => eq(k.id, keyId) } as any),
        )
        if (!key)
          return null

        // Ownership-OR-membership guard.
        if (key.reference === 'user') {
          if (String(key.referenceId) !== String(user.id))
            return null
          return key
        }
        // organization scope — caller must be a member.
        const isMember = await ctx.runEffect(
          Effect.gen(function* () {
            const org = yield* OrganizationService
            return yield* org.checkMembership(key.referenceId, Number(user.id))
          }),
        )
        return isMember ? key : null
      },
    }))

  // ── myApiKeys — all keys owned by the calling user ────────────────────────
  builder.queryField('myApiKeys', t =>
    t.drizzleField({
      type: ['apikeys'],
      authScopes: { auth: true },
      resolve: async (query, _root: unknown, _args: unknown, ctx: GraphQLContextMap) => {
        const user = ctx.auth?.user
        if (!user)
          throw new UnauthenticatedError()

        const db = await ctx.runEffect(DrizzleDb) as any
        return db.query.apikeys.findMany(query({
          where: (k: any, { eq, and }: any) => and(eq(k.reference, 'user'), eq(k.referenceId, Number(user.id))),
        } as any))
      },
    }))

  // ── organizationApiKeys — all keys owned by a given organization ──────────
  builder.queryField('organizationApiKeys', t =>
    t.drizzleField({
      type: ['apikeys'],
      args: {
        organizationId: t.arg.id({ required: true }),
      },
      authScopes: (_parent: unknown, args: any) => ({
        permission: {
          resource: 'api-key',
          actions: ['read'],
          organization: Number(decodeGlobalID(args.organizationId).id),
        },
      }),

        } as any))
      },
    }))
```
